class Menu

Attributes:

menu_items = (hash)
menu_color = (string)
are_specials? = (boolean)
amount_ordered = (float)

Methods:

change_color = re-assign string for menu_color
remove_item = remove an item from the menu_items array
check_deals = if are_specials? == true, menu_items[:"*name of item"] = new price
add_item = adds an item to the menu hash
order_total = menu_items[:"*name of menu item"] * amount_ordered, return total
